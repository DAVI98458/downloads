-- [ SISTEMA DE ARRASTAR ] --
local dragging = false
local dragInput = nil
local dragStart = nil
local startPos = nil

header.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        dragging = true
        dragStart = input.Position
        startPos = mainFrame.Position
        TweenService:Create(mainFrame, TweenInfo.new(0.1), {Size = UDim2.new(0, 415, 0, 275)}):Play()
    end
end)

header.InputEnded:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        dragging = false
        TweenService:Create(mainFrame, TweenInfo.new(0.1), {Size = UDim2.new(0, 420, 0, 280)}):Play()
    end
end)

UserInputService.InputChanged:Connect(function(input)
    if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
        local delta = input.Position - dragStart
        mainFrame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
    end
end)

-- Hover effects
local function addHoverEffect(button, normalColor, hoverColor)
    button.MouseEnter:Connect(function()
        TweenService:Create(button, TweenInfo.new(0.2), {BackgroundColor3 = hoverColor}):Play()
    end)
    button.MouseLeave:Connect(function()
        TweenService:Create(button, TweenInfo.new(0.2), {BackgroundColor3 = normalColor}):Play()
    end)
end

-- [ FUNÇÃO PRINCIPAL WINDUI ] --
function LoadWindUI()
    local WindUI = loadstring(game:HttpGet("https://github.com/Footagesus/WindUI/releases/latest/download/main.lua"))()

    local lp = game.Players.LocalPlayer
    local Circle = Drawing.new("Circle") 
    local Tracers = {}
    _G.InfJump = false
    _G.LockAssassino = false
    _G.TracersVisible = false
    _G.ESPVisible = false
    _G.AntiAFK = true
    _G.HB = false
    _G.KillAuraCircle = false

    -- [ CÍRCULO VERMELHO - SÓ BORDA, SEM FILL ] --
    Circle.Visible = false
    Circle.Radius = 270 
    Circle.Thickness = 1.5  -- Borda mais grossa pra destacar
    Circle.Color = Color3.fromRGB(255, 0, 0)  -- VERMELHO PURO
    Circle.Filled = false   -- SEM FILL - só a borda!
    Circle.Transparency = 1

    -- [ TRACERS BRANCOS LITERALMENTE ] --
    local function CreateTracer(player)
        if player == lp then return end 
        local line = Drawing.new("Line")
        line.Visible = false
        line.Color = Color3.fromRGB(255, 255, 255)  -- BRANCO LITERAL
        line.Thickness = 1.5
        Tracers[player] = line
    end

    local function RemoveTracer(player)
        if Tracers[player] then
            Tracers[player].Visible = false
            Tracers[player]:Remove()
            Tracers[player] = nil
        end
    end

    for _, v in pairs(game.Players:GetPlayers()) do CreateTracer(v) end
    game.Players.PlayerAdded:Connect(CreateTracer)
    game.Players.PlayerRemoving:Connect(RemoveTracer)

    -- Pulo infinito
    game:GetService("UserInputService").JumpRequest:Connect(function()
        if _G.InfJump and lp.Character and lp.Character:FindFirstChildOfClass("Humanoid") then
            lp.Character:FindFirstChildOfClass("Humanoid"):ChangeState("Jumping")
        end
    end)

    -- Loop principal
    game:GetService("RunService").RenderStepped:Connect(function()
        local camSize = workspace.CurrentCamera.ViewportSize
        Circle.Position = Vector2.new(camSize.X / 2, camSize.Y / 2)

        local char = lp.Character
        local knife = char and char:FindFirstChild("Knife")
        local gun = char and char:FindFirstChild("Gun")

        for _, p in pairs(game.Players:GetPlayers()) do
            if p ~= lp and p.Character then
                -- [ ESP COM HIGHLIGHT - VERDE BASE, CORES ESPECIAIS PARA ARMAS ] --
                local highlight = p.Character:FindFirstChild("CydexHighlight")
                if _G.ESPVisible then
                    if not highlight then
                        highlight = Instance.new("Highlight", p.Character)
                        highlight.Name = "CydexHighlight"
                    end
                    highlight.Enabled = true
                    highlight.OutlineTransparency = 0.2  -- Outline no 0.3
                    highlight.FillTransparency = 1       -- Fill no 0 (totalmente preenchido)

                    -- Verificar se tem faca ou gun primeiro
                    local hasKnife = p.Backpack:FindFirstChild("Knife") or p.Character:FindFirstChild("Knife")
                    local hasGun = p.Backpack:FindFirstChild("Gun") or p.Character:FindFirstChild("Gun")

                    if hasKnife then
                        highlight.FillColor = Color3.fromRGB(255, 0, 0)  -- Vermelho para faca
                        highlight.OutlineColor = Color3.fromRGB(255, 50, 50)
                    elseif hasGun then
                        highlight.FillColor = Color3.fromRGB(0, 0, 255)  -- Azul para gun
                        highlight.OutlineColor = Color3.fromRGB(50, 50, 255)
                    else
                        highlight.FillColor = Color3.fromRGB(0, 255, 0)  -- Verde para inocente
                        highlight.OutlineColor = Color3.fromRGB(50, 255, 50)
                    end
                elseif highlight then 
                    highlight.Enabled = false 
                end

                -- Hitbox
                if _G.HB and p.Character:FindFirstChild("HumanoidRootPart") then
                    p.Character.HumanoidRootPart.Size = Vector3.new(27, 27, 27)
                    p.Character.HumanoidRootPart.Transparency = 1
                    p.Character.HumanoidRootPart.CanCollide = false
                end

                -- [ TRACERS BRANCOS ] --
                local line = Tracers[p]
                if line then
                    if _G.TracersVisible and p.Character:FindFirstChild("HumanoidRootPart") then
                        local loc, onScreen = workspace.CurrentCamera:WorldToViewportPoint(p.Character.HumanoidRootPart.Position)
                        if onScreen then
                            line.From = Vector2.new(camSize.X / 2, camSize.Y - 10)
                            line.To = Vector2.new(loc.X, loc.Y)
                            line.Visible = true
                        else 
                            line.Visible = false 
                        end
                    else 
                        line.Visible = false 
                    end
                end

                -- Kill Aura Circle (círculo vermelho)
                if _G.KillAuraCircle and knife and p.Character:FindFirstChild("HumanoidRootPart") then
                    local pos, onScreen = workspace.CurrentCamera:WorldToViewportPoint(p.Character.HumanoidRootPart.Position)
                    if onScreen then
                        local dist = (Vector2.new(pos.X, pos.Y) - Circle.Position).Magnitude
                        if dist <= Circle.Radius then
                            workspace.CurrentCamera.CFrame = CFrame.new(workspace.CurrentCamera.CFrame.Position, p.Character.HumanoidRootPart.Position)
                        end
                    end
                end
            end
        end

        -- Lock no assassino
        if _G.LockAssassino and gun then
            for _, p in pairs(game.Players:GetPlayers()) do
                if p ~= lp and p.Character and (p.Backpack:FindFirstChild("Knife") or p.Character:FindFirstChild("Knife")) then
                    workspace.CurrentCamera.CFrame = CFrame.new(workspace.CurrentCamera.CFrame.Position, p.Character.UpperTorso.Position)
                end
            end
        end
    end)

    -- Criar janela WindUI
    local Window = WindUI:CreateWindow({
        Title = "SCRIPT MM2 OP (Ator)",
        Author = "by Shadown_gd and guii_ntx7",
        Folder = "CydexMM2",
        Size = UDim2.fromOffset(550, 400),
        Transparent = true,
        Theme = "Dark",
        Resizable = true,
        SideBarWidth = 200,
    })

    -- Aplicar tema com pcall
    task.delay(1, function()
        local success = pcall(function()
            WindUI:AddTheme({
                Name = "PurpleTheme",
                Accent = Color3.fromHex("#8b5cf6"),
                Dialog = Color3.fromHex("#6d28d9"),
                Outline = Color3.fromHex("#a78bfa"),
                Text = Color3.fromHex("#ffffff"),
                Placeholder = Color3.fromHex("#c4b5fd"),
                Button = Color3.fromHex("#7c3aed"),
                Icon = Color3.fromHex("#a78bfa"),
                WindowBackground = Color3.fromHex("#4c1d95"),
                TopbarButtonIcon = Color3.fromHex("#a78bfa"),
                TopbarTitle = Color3.fromHex("#ffffff"),
                TopbarAuthor = Color3.fromHex("#ffffff"),
                TopbarIcon = Color3.fromHex("#8b5cf6"),
                TabBackground = Color3.fromHex("#5b21b6"),
                TabTitle = Color3.fromHex("#ffffff"),
                TabIcon = Color3.fromHex("#a78bfa"),
                ElementBackground = Color3.fromHex("#7c3aed"),
                ElementTitle = Color3.fromHex("#ffffff"),
                ElementDesc = Color3.fromHex("#ffffff"),
                ElementIcon = Color3.fromHex("#a78bfa"),
            })
            WindUI:SetTheme("PurpleTheme")
        end)
        if not success then warn("[CYDEX] Erro ao aplicar tema") end
    end)

    -- Abas
    local HomeTab = Window:Tab({Title = "Home", Icon = "info"})
    local TrollTab = Window:Tab({Title = "Troll & Combat", Icon = "sword"})
    local AntiTab = Window:Tab({Title = "Sistema Anti", Icon = "shield"})
    local LagTab = Window:Tab({Title = "Anti-Lag", Icon = "zap"})
    local InfoTab = Window:Tab({Title = "Informações", Icon = "info"})

    -- Home
    HomeTab:Image({
        Name = "Logo",
        Image = "rbxassetid://81424275303971",
        Size = UDim2.new(1, -20, 0, 200),
        Position = UDim2.new(0, 10, 0, 10),
    })
    HomeTab:Paragraph({
        Title = "Bem-vindo ao SCRIPT MM2 OP",
        Desc = "Use as abas laterais para acessar as funcionalidades."
    })

    -- Troll & Combat
    TrollTab:Toggle({
        Title = "Kill Aura no Círculo",
        Value = false,
        Callback = function(Value) _G.KillAuraCircle = Value end
    })
    TrollTab:Toggle({
        Title = "Círculo de Mira (280) - Vermelho",

    })
    TrollTab:Toggle({

    TrollTab:Toggle({

                lp.Character.Humanoid.WalkSpeed = Value and 25 or 16

        end
    })
    TrollTab:Toggle({
        Title = "Pulo Infinito",
        Value = false,
        Callback = function(Value) _G.InfJump = Value end
    })

    -- Sistema Anti
    AntiTab:Toggle({
        Title = "Anti-Denúncia & Ban",
        Value = true,
        Callback = function() end
    })
    AntiTab:Toggle({
        Title = "Anti-Aviso 267",

    })

    })
    AntiTab:Toggle({
        Title = "Anti-AFK",
        Value = true,
        Callback = function(Value) _G.AntiAFK = Value end
    })

    -- Anti-Lag
    LagTab:Toggle({
        Title = "Remover Travamentos (Mobile)",
        Value = false,
