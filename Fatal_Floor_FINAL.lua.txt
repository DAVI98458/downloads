-- Function to find ProximityPrompts
local function findOreoPrompts()
    local prompts = {}
    local procGenItems = workspace:FindFirstChild("ProcGenGenerated") and workspace.ProcGenGenerated:FindFirstChild("ProcGenItems")

    if procGenItems then
        for _, item in ipairs(procGenItems:GetDescendants()) do
            if item:IsA("ProximityPrompt") then
                table.insert(prompts, item)
            end
        end
    end

    return prompts
end

-- Function to teleport and activate
local function teleportAndActivate(prompt)
    local character = game.Players.LocalPlayer.Character
    if not character or not character:FindFirstChild("HumanoidRootPart") then return end

    local hrp = character.HumanoidRootPart
    local parent = prompt.Parent

    if parent and parent:IsA("BasePart") then
        hrp.CFrame = parent.CFrame + Vector3.new(0, 3, 0)
        task.wait(0.2)
        fireproximityprompt(prompt)
        task.wait(0.3)

        -- Volta para posição fixa: -2, 38, 11
        hrp.CFrame = CFrame.new(-2, 38, 11)
    end
end

-- Button Grab 1 Oreo
CollectTab:Button({
    Title = "Grab a Ore",
    Desc = "This one only picks up 1 ore",
    Callback = function()
        local prompts = findOreoPrompts()

        if #prompts > 0 then
            teleportAndActivate(prompts[1])

                Title = "Error",

})

-- AUTO FARM V1 LOOP
task.spawn(function()
    while true do
        if not _G.FatalFloorAutoFarmV1Enabled then
            _G.FatalFloorAutoFarmV1Active = false
            task.wait(0.2)
            continue
        end

        if _G.FatalFloorAutoFarmV1Active then
            task.wait(0.2)
            continue
        end

        _G.FatalFloorAutoFarmV1Active = true

        while _G.FatalFloorAutoFarmV1Enabled do
            local player = game.Players.LocalPlayer
            local character = player.Character
            if not character then task.wait(0.5) continue end

            local humanoid = character:FindFirstChildOfClass("Humanoid")
            if not humanoid then task.wait(0.5) continue end

            local backpack = player:FindFirstChild("Backpack")
            local hrp = character:FindFirstChild("HumanoidRootPart")
            if not hrp then task.wait(0.5) continue end

            local prompts = findOreoPrompts()
            if #prompts == 0 then
                task.wait(0.5)
                continue
            end

            local parent = prompts[1].Parent
            if parent and parent:IsA("BasePart") then
                hrp.CFrame = parent.CFrame + Vector3.new(0, 3, 0)
                task.wait(0.2)
                fireproximityprompt(prompts[1])
                task.wait(0.3)
            end

            -- Volta para posição fixa: -2, 38, 11
            hrp.CFrame = CFrame.new(-2, 38, 11)

            task.wait(0.5)
            if not _G.FatalFloorAutoFarmV1Enabled then break end

            local holding = false
            for _, child in ipairs(character:GetChildren()) do
                if child:IsA("Tool") then
                    holding = true
                    break
                end
            end

            if not holding then
                local tools = {}
                if backpack then
                    for _, item in ipairs(backpack:GetChildren()) do
                        if item:IsA("Tool") then
                            table.insert(tools, item)
                        end
                    end
                end

                if #tools > 0 then
                    pcall(function()
                        humanoid:EquipTool(tools[1])
                    end)
                    task.wait(0.2)

                    local stillNotHolding = true
                    for _, child in ipairs(character:GetChildren()) do
                        if child:IsA("Tool") then
                            stillNotHolding = false
                            break
                        end
                    end

                    if stillNotHolding then
                        pcall(function()
                            tools[1].Parent = character
                        end)
                        task.wait(0.2)
                    end
                end
            end

            local hasTool = false
            for _, child in ipairs(character:GetChildren()) do
                if child:IsA("Tool") then
                    hasTool = true
                    break
                end
            end

            if not hasTool then
                task.wait(0.5)
                continue
            end

            local sellSpot = workspace:FindFirstChild("PlacedTiles") 
                and workspace.PlacedTiles:FindFirstChild("SellStation") 
                and workspace.PlacedTiles.SellStation:FindFirstChild("SellingSpot")

            if not sellSpot or not sellSpot:IsA("BasePart") then
                task.wait(0.5)
                continue
            end

            hrp.CFrame = sellSpot.CFrame + Vector3.new(0, 3, 0)
            task.wait(0.3)
            if not _G.FatalFloorAutoFarmV1Enabled then break end

            for i = 1, 20 do
                if not _G.FatalFloorAutoFarmV1Enabled then break end

                for _, d in ipairs(sellSpot:GetDescendants()) do
                    if d:IsA("ProximityPrompt") then
                        fireproximityprompt(d)
                    end
                end

                task.wait(0.1)

                local stillHasTool = false
                for _, child in ipairs(character:GetChildren()) do
                    if child:IsA("Tool") then
                        stillHasTool = true
                        break
                    end
                end

                if not stillHasTool then
                    break
                end
            end

            if not _G.FatalFloorAutoFarmV1Enabled then break end

            -- Volta para posição fixa: -2, 38, 11
            hrp.CFrame = CFrame.new(-2, 38, 11)

            task.wait(0.5)
        end

        _G.FatalFloorAutoFarmV1Active = false
    end
end)

local EspTab = Window:Tab({
    Title = "Esp",
    Icon = "eye",
})

EspTab:Section({
    Title = "Monster ESP"
})

local espHighlights = {}
local espMonstersEnabled = false

local function clearEsp()
    for _, highlight in ipairs(espHighlights) do
        if highlight then
            highlight:Destroy()
        end
    end
    espHighlights = {}
end

local function createEspForMonster(monster)
    if monster.Name == "PlantMonster" then return end

    local hrp = monster:FindFirstChild("HumanoidRootPart") or monster:FindFirstChildWhichIsA("BasePart")
    if not hrp then return end

    for _, h in ipairs(espHighlights) do
        if h and h.Parent == monster then
            return
        end
    end

    local highlight = Instance.new("Highlight")
    highlight.Name = "MonsterESP"
    highlight.Adornee = monster
    highlight.FillColor = Color3.fromRGB(255, 0, 0)
    highlight.OutlineColor = Color3.fromRGB(255, 0, 0)
    highlight.FillTransparency = 1
    highlight.OutlineTransparency = 0.3
    highlight.Parent = monster

    table.insert(espHighlights, highlight)
end

EspTab:Toggle({
    Title = "Esp Monsters",
    Desc = "Highlight all monsters in workspace.Monsters (red, outline 0.3, fill 1). Ignores PlantMonster.",
    Callback = function(state)
        espMonstersEnabled = state

        if state then
            WindUI:Notify({
                Title = "ESP",
                Content = "ESP Monsters enabled!",
                Duration = 3,
            })

            local monstersFolder = workspace:FindFirstChild("Monsters")
            if monstersFolder then
                for _, monster in ipairs(monstersFolder:GetChildren()) do
                    if monster:IsA("Model") or monster:IsA("BasePart") then
                        createEspForMonster(monster)
                    end
                end
            end

            task.spawn(function()
                while espMonstersEnabled do
                    local monstersFolder = workspace:FindFirstChild("Monsters")
                    if monstersFolder then
                        for _, monster in ipairs(monstersFolder:GetChildren()) do
                            if monster:IsA("Model") or monster:IsA("BasePart") then
                                createEspForMonster(monster)
                            end
                        end
                    end

                    for i = #espHighlights, 1, -1 do
                        local h = espHighlights[i]
                        if not h or not h.Parent then
                            table.remove(espHighlights, i)
                        end
                    end

                    task.wait(0.5)
                end
            end)
        else
            clearEsp()
            WindUI:Notify({
                Title = "ESP",
                Content = "ESP Monsters disabled!",
                Duration = 3,
            })
        end
    end,
})

local AntiTab = Window:Tab({
    Title = "Anti",
    Icon = "shield",
})

AntiTab:Section({
    Title = "Monster Defense"
})

local antiBlobEnabled = false

AntiTab:Toggle({
    Title = "anti-blob",
    Desc = "Delete the monster (FE)",
    Callback = function(state)
        antiBlobEnabled = state

        if state then
            WindUI:Notify({
                Title = "Anti-Blob",
                Content = "Anti-blob enabled!",
                Duration = 3,
            })

            task.spawn(function()
                while antiBlobEnabled do
                    local monstersFolder = workspace:FindFirstChild("Monsters")
                    local deletedCount = 0

                    if monstersFolder then
                        for _, child in ipairs(monstersFolder:GetChildren()) do
                            if child.Name == "FleshMonster" then
                                local hrp = child:FindFirstChild("HumanoidRootPart")
                                if hrp then
                                    hrp:Destroy()
                                    deletedCount = deletedCount + 1
                                end
                            end
                        end
                    end

                    if deletedCount > 0 then
                        WindUI:Notify({
                            Title = "Anti-Blob",
                            Content = "Deleted " .. deletedCount .. " HumanoidRootPart!",
                            Duration = 1,
                        })
                    end

                    task.wait(0.1)
                end
            end)
        else
            WindUI:Notify({
                Title = "Anti-Blob",
                Content = "Anti-blob disabled!",
                Duration = 3,
            })
        end
    end,
})
